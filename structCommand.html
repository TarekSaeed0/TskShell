Command Syntax

command = word , { word }, [ & ] ; word = { word_element } ;

word_element = unquoted_character | single_quoted | double_quoted | expansion ;

unquoted_character = ? any character except whitespace, ', ", & and $ ? ; single_quoted = "'" , { single_quoted_character } , "'" ; double_quoted = '"' , { double_quoted_character | expansion } , '"' ;

single_quoted_character = ? any character except ' ? ; double_quoted_character = ? any character except " and $ ? ; expansion = "$" , ( identifier | "{" , identifier , "}" ) ; identifier = letter | "_", { letter | digit | "_" } ; letter = "a".."z" | "A".."Z" ; digit = "0".."9" ;
